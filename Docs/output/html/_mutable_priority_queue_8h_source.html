122void MutablePriorityQueue::heapifyDown(unsigned i) {
123 auto x = H[i];
124 while (true) {
125 unsigned k = leftChild(i);
126 if (k >= H.size())
127 break;
128 if (k+1 < H.size() && *H[k+1] < *H[k])
129 ++k; // right child of i

131 break;
132 set(i, H[k]);
133 i = k;
134 }
